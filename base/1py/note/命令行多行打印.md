

```py
import time 
import sys

#初始行
sys.stdout.write("#\n")
sys.stdout.write("#")
sys.stdout.flush()

for i in range(10):
	sys.stdout.write("\033[1A")		#光标上移1行
	sys.stdout.write("\r")

	m = 0
	while m < i:
		sys.stdout.write("%")	#不换行输出
		m = m + 1
	sys.stdout.write("\033[1B")	#光标下移一行
	
	sys.stdout.write("\r")
	m = 0
	while m < i+1:
		sys.stdout.write("#")	#不换行输出
		m = m + 1

	sys.stdout.flush()			#立即刷新
	time.sleep(0.1)


sys.stdout.write("\nDone")
```
